var loadouts = import my/loadouts

func equip_melee()
  equipL sword *10
  equipR shield *10 fire

func equip_dashing()
  equipL sword *10
  equipR shield *10 dashing

func equip_staff()
  equip staff *8 ice

func equip_ranged_healing()
  equipL crossbow *10
  equipR ouroboros

?loc = caves
  ?foe ! boss  // Small enemies
    ?hp = maxhp
      equip_staff()
    :
      equip_ranged_healing()
  :?foe = boss  // Bolesh, the cunning
    ?foe.distance > 10
      equip_dashing()
    :
      equip_melee()
  :?hp < 10
    activate potion

  loadouts.healwalk()
